/////BOUCLE DE JEU

    FONCTION boucleJeu()
        effacer le canva 

        SI ETAT == "intro" ALORS
            Dessiner le menu d'accueil
        SINON SI ETAT == "jeu" ALORS 
            Dessiner les questions
            Dessiner le score (UI)
        FIN SI

    FIN FONCTION    


## FONCTIONS D'ÉCOUTER DES ÉVÉNEMENTS

FUNCTION onClicCanvas(evenement)
    Récupérer la position du curseur ( X, Y)

    SI ETAT == "intro" && clic  sur le bouton "Démarrer" ALORS
        ETAT = "jeu"

    POUR la position de question change d 1 jusqu'à 
        il n'en restent plus de questions ALORS
        BOUTON = ALORS
        Faut calculer la position du bouton dans le canva   

        COLLITION = detecterClicObjet( curseurX, curseurY, bouton);
        SI COLLITION == true ALORS
            PREMIERE_QUESTION = LISTE_QUESTIONS[INDEX_QUIZ];
            
            Si PREMIERE_QUESTION.reponse == premiereQuestion.choix[i] ALORS
                AFFICHER message de reussite;
                Afficher sons de reussite;
                Augmenter npoints + 1;
            } SINON {
                AFFICHER message de defaite; 
                Afficher sons d'erreur;
                Diminuer npoints - 1;
            FIN SI    

            INDEX_QUIZ += 1;
            SI INDEX_QUIZ >=  question.length ALORS
                ETAT = "fin"
            FIN SI
        FIN SI
    FIN POUR

FIN FONCTION
